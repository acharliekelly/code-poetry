Chickens

eggs: Date[]

      function birdCount(eggs) {
        let chickens = 0;
        const today = new Date();
        for (let i=0; i<=eggs.length; i++) {
          if (eggs[i] <= today) {
            // egg is hatched
            chickens++;
          } else {
            // do not count
          }
        }
        return chickens;
      }
